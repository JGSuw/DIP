00051 void arm_cmplx_mag_squared_q15(     
00052   q15_t * pSrc,     
00053   q15_t * pDst,     
00054   uint32_t numSamples)     
00055 {     
00056   q31_t in1, in2, in3, in4;                      /* Temporary variables to hold input data */  
00057   q31_t acc1, acc2, acc3, acc4;                  /* Temporary accumulators */  
00058   q15_t out1, out2, out3, out4;                  /* Temporary variables to hold output data */  
00059   uint32_t blkCnt;                               /* loop counter */     
00060      
00061   /*loop Unrolling */     
00062   blkCnt = numSamples >> 2u;     
00063      
00064   /* First part of the processing with loop unrolling.  Compute 4 outputs at a time.      
00065    ** a second loop below computes the remaining 1 to 3 samples. */     
00066   while(blkCnt > 0u)     
00067   {    
00068     /* read complex input from source buffer */  
00069     in1 = _SIMD32_OFFSET(pSrc);  
00070     in2 = _SIMD32_OFFSET(pSrc + 2);  
00071     in3 = _SIMD32_OFFSET(pSrc + 4);  
00072     in4 = _SIMD32_OFFSET(pSrc + 6);  
00073   
00074     /* calculate powers of real and imaginary values and accumulate */  
00075     acc1 = __SMUAD(in1, in1);  
00076     acc2 = __SMUAD(in2, in2);  
00077     acc3 = __SMUAD(in3, in3);  
00078     acc4 = __SMUAD(in4, in4);  
00079   
00080     /* get the output to 3.13 format */  
00081     out1 = (q15_t)(acc1 >> 17);  
00082     out2 = (q15_t)(acc2 >> 17);  
00083     out3 = (q15_t)(acc3 >> 17);  
00084     out4 = (q15_t)(acc4 >> 17);  
00085   
00086     /* packed output data for SIMD store */  
00087 #ifndef ARM_MATH_BIG_ENDIAN  
00088   
00089     acc1 = __PKHBT(out1, out2, 16);  
00090     acc2 = __PKHBT(out3, out4, 16);  
00091   
00092 #else  
00093       
00094     acc1 = __PKHBT(out2, out1, 16);  
00095     acc2 = __PKHBT(out4, out3, 16);  
00096   
00097 #endif      //  #ifndef ARM_MATH_BIG_ENDIAN     
00098   
00099     /* store result to destination buffer */  
00100     _SIMD32_OFFSET(pDst) = acc1;  
00101     _SIMD32_OFFSET(pDst + 2) = acc2;  
00102   
00103     /* increment source pointer by 8 to process next samples */  
00104     pSrc += 8u;  
00105     /* increment destination pointer by 4 to process next samples */  
00106     pDst += 4u;  
00107   
00108     /* Decrement the loop counter */     
00109     blkCnt--;     
00110   }     
00111      
00112   /* If the numSamples is not a multiple of 4, compute any remaining output samples here.      
00113    ** No loop unrolling is used. */     
00114   blkCnt = numSamples % 0x4u;     
00115      
00116   while(blkCnt > 0u)     
00117   {     
00118     /* C[0] = (A[0] * A[0] + A[1] * A[1]) */     
00119     in1 = *__SIMD32(pSrc)++;

00121   
00122     out1 = (q15_t)(acc1 >> 17);  
00123       
00124     /* store the result in 3.13 format in the destination buffer. */     
00125     *pDst++ = out1;     
00126      
00127     /* Decrement the loop counter */     
00128     blkCnt--;
